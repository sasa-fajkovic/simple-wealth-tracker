**Click-outside handler** — use `useRef` + `useEffect`:
```typescript
const menuRef = useRef<HTMLDivElement>(null)
useEffect(() => {
  if (!open) return
  const handler = (e: MouseEvent) => {
    if (menuRef.current && !menuRef.current.contains(e.target as Node)) {
      setOpen(false)
    }
  }
  document.addEventListener('mousedown', handler)
  return () => document.removeEventListener('mousedown', handler)
}, [open])
```
Attach `ref={menuRef}` to the outer `<nav>` element (the element wrapping both the bar and the dropdown).

**Mobile dropdown NavLinks** — each must call `onClick={() => setOpen(false)}` so the dropdown closes on navigation.

Complete file:
```typescript
import { useState, useEffect, useRef } from 'react'
import { NavLink } from 'react-router-dom'
import { Menu, X } from 'lucide-react'

export default function Nav() {
  const [open, setOpen] = useState(false)
  const menuRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    if (!open) return
    const handler = (e: MouseEvent) => {
      if (menuRef.current && !menuRef.current.contains(e.target as Node)) {
        setOpen(false)
      }
    }
    document.addEventListener('mousedown', handler)
    return () => document.removeEventListener('mousedown', handler)
  }, [open])

  const linkClass = ({ isActive }: { isActive: boolean }) =>
    isActive
      ? 'text-blue-600 font-medium text-sm transition-colors'
      : 'text-gray-500 font-medium text-sm hover:text-gray-900 transition-colors'

  return (
    <nav
      ref={menuRef as React.RefObject<HTMLElement>}
      className="h-14 bg-white border-b border-gray-200 px-4 sm:px-8 flex items-center justify-between relative"
    >
      {/* Logo */}
      <NavLink to="/" className="text-lg font-medium text-gray-900">
        WealthTrack
      </NavLink>

      {/* Desktop links */}
      <div className="hidden sm:flex gap-6">
        <NavLink to="/" end className={linkClass}>Dashboard</NavLink>
        <NavLink to="/projections" className={linkClass}>Projections</NavLink>
        <NavLink to="/admin" className={linkClass}>Admin</NavLink>
      </div>

      {/* Hamburger button — mobile only */}
      <button
        className="sm:hidden p-2 rounded text-gray-500 hover:text-gray-900 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-offset-2"
        aria-label={open ? 'Close navigation menu' : 'Open navigation menu'}
        onClick={() => setOpen(o => !o)}
      >
        {open ? <X size={20} /> : <Menu size={20} />}
      </button>

      {/* Mobile dropdown */}
      {open && (
        <div className="absolute top-14 left-0 right-0 bg-white border-b border-gray-200 z-50 flex flex-col py-2">
          <NavLink to="/" end className={({ isActive }) =>
            `px-4 py-3 text-sm font-medium ${isActive ? 'text-blue-600' : 'text-gray-500 hover:text-gray-900'}`
          } onClick={() => setOpen(false)}>Dashboard</NavLink>
          <NavLink to="/projections" className={({ isActive }) =>
            `px-4 py-3 text-sm font-medium ${isActive ? 'text-blue-600' : 'text-gray-500 hover:text-gray-900'}`
          } onClick={() => setOpen(false)}>Projections</NavLink>
          <NavLink to="/admin" className={({ isActive }) =>
            `px-4 py-3 text-sm font-medium ${isActive ? 'text-blue-600' : 'text-gray-500 hover:text-gray-900'}`
          } onClick={() => setOpen(false)}>Admin</NavLink>
        </div>
      )}
    </nav>
  )
}
```

Note: `ref={menuRef}` on the `<nav>` element requires casting — use `ref={menuRef as React.RefObject<HTMLElement>}` since HTMLElement is the type of `<nav>`. This is correct and TypeScript-safe.
