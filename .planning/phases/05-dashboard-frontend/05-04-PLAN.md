```typescript
import type { SummaryResponse } from '../types/index'

// Instantiated once at module scope — do not recreate per render
const eurFormatter = new Intl.NumberFormat('de-DE', { style: 'currency', currency: 'EUR' })

function formatDelta(abs: number, pct: number): { text: string; positive: boolean } {
  const prefix = abs >= 0 ? '+' : '\u2212'  // U+2212 MINUS SIGN, not hyphen
  const absPct = Math.abs(pct).toFixed(2)
  const pctPrefix = abs >= 0 ? '+' : '\u2212'
  return {
    text: `${prefix}${eurFormatter.format(Math.abs(abs))} (${pctPrefix}${absPct}%)`,
    positive: abs >= 0,
  }
}

interface SummaryCardsProps {
  data: SummaryResponse
}

export default function SummaryCards({ data }: SummaryCardsProps) {
  const { current_total, period_delta_abs, period_delta_pct, category_breakdown } = data
  const delta = formatDelta(period_delta_abs, period_delta_pct)

  return (

      {/* Card 1 — Total Net Worth */}
      <div className="bg-white rounded-lg border border-gray-200 p-6">
        <p className="text-xs font-medium text-gray-500 uppercase tracking-wide mb-1">
          Total Net Worth
        </p>
        <p className="text-2xl font-bold text-gray-900">
          {eurFormatter.format(current_total)}
        </p>
      </div>

      <div className="bg-white rounded-lg border border-gray-200 p-6">
        <p className="text-xs font-medium text-gray-500 uppercase tracking-wide mb-1">
          Period Change
        </p>
        <p className={`text-sm font-medium ${delta.positive ? 'text-green-600' : 'text-red-600'}`}>
          {delta.text}
        </p>
      </div>

      {/* Card 3 — Category Breakdown */}
      <div className="bg-white rounded-lg border border-gray-200 p-6">
        <p className="text-xs font-medium text-gray-500 uppercase tracking-wide mb-3">
          Category Breakdown
        </p>
        <div className="flex flex-col gap-2">

              <span className="flex items-center gap-2">
                <span
                  className="w-2 h-2 rounded-full flex-shrink-0"
                  style={{ backgroundColor: row.color }}
                />
                <span className="text-gray-700">{row.category_name}</span>
              </span>
              <span className="text-gray-900">{eurFormatter.format(row.value)}</span>
```

```typescript
import { useState, useEffect } from 'react'
import { getSummary, ApiError } from '../api/client'
import type { SummaryResponse } from '../types/index'
import type { RangeKey } from '../types/index'
import ChartTypeSelector, { useChartType } from '../components/ChartTypeSelector'
import WealthChart from '../components/WealthChart'
import SummaryCards from '../components/SummaryCards'

const RANGES: { label: string; value: RangeKey }[] = [
  { label: 'YTD', value: 'ytd' },
  { label: '6M',  value: '6m'  },
  { label: '1Y',  value: '1y'  },
  { label: '2Y',  value: '2y'  },
  { label: '3Y',  value: '3y'  },
  { label: '5Y',  value: '5y'  },
  { label: '10Y', value: '10y' },
  { label: 'Max', value: 'max' },
]

export default function DashboardPage() {
  const [range, setRange] = useState<RangeKey>('1y')
  const [data, setData] = useState<SummaryResponse | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const [retryCount, setRetryCount] = useState(0)
  const [chartType, setChartType] = useChartType('dashboard-chart-type')

  // Set page title on mount
  useEffect(() => {
    document.title = 'Dashboard — WealthTrack'
  }, [])

  // Fetch summary data whenever range or retryCount changes
  useEffect(() => {
    let cancelled = false
    setLoading(true)
    setError(null)
    getSummary(range)
      .then(result => {
        if (!cancelled) setData(result)
      })
      .catch(err => {
        if (!cancelled) {
          setError(err instanceof ApiError ? err.message : 'Unexpected error')
        }
      })
      .finally(() => {
        if (!cancelled) setLoading(false)
      })
    return () => { cancelled = true }
  }, [range, retryCount])

  return (
    <div className="min-h-screen bg-gray-50">
      <div className="max-w-7xl mx-auto px-4 sm:px-8 py-8">

        {/* Controls row: range pills (left) + chart type selector (right) */}
        <div className="flex flex-wrap items-center justify-between gap-3 mb-4">
          <div className="flex flex-wrap gap-1">
            {RANGES.map(r => (
              <button
                key={r.value}
                onClick={() => setRange(r.value)}
                className={
                  range === r.value
                    ? 'bg-blue-600 border border-blue-600 text-white rounded-full px-3 py-1 text-xs font-medium cursor-pointer'
                    : 'bg-white border border-gray-200 text-gray-600 hover:bg-gray-50 rounded-full px-3 py-1 text-xs font-medium transition-colors cursor-pointer'
                }
              >
                {r.label}
              </button>
            ))}
          </div>
          <ChartTypeSelector value={chartType} onChange={setChartType} />
        </div>

        {/* Error banner */}
        {error && (
          <div className="bg-red-50 border border-red-200 rounded-lg p-4 mb-6">
            <p className="text-red-700 font-medium text-sm">Could not load data</p>
            <p className="text-red-600 text-sm mt-1">Check your connection and refresh the page.</p>
            <button
              onClick={() => setRetryCount(c => c + 1)}
              className="mt-2 text-sm font-medium text-red-700 underline"
            >
              Retry loading data
            </button>
          </div>
        )}

        {/* Loading skeleton */}
        {loading ? (
          <>
            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4 mb-6">
              {[0, 1, 2].map(i => (
                <div key={i} className="h-24 bg-gray-100 rounded-lg animate-pulse" />
              ))}
            </div>
            <div className="h-[420px] bg-gray-100 rounded-lg animate-pulse" />
          </>
        ) : data ? (
          <>
            <SummaryCards data={data} />
            <WealthChart data={data} chartType={chartType} />
          </>
        ) : null}

      </div>
    </div>
  )
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| getSummary API response → React state | Financial totals and category data flows from server into component state and DOM |
| User click (range pill / retry) → fetch | User-triggered actions initiate network requests |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-12 | XSS | SummaryCards — category_name rendered in JSX spans | mitigate | React JSX renders all dynamic strings as text nodes (not HTML). `dangerouslySetInnerHTML` is never used. Enforced by code pattern |
| T-05-13 | XSS | SummaryCards — `row.color` in `style={{ backgroundColor: ... }}` | accept | CSS property injection in modern browsers cannot execute JavaScript via style attribute. No `expression()` support |
| T-05-14 | Spoofing | error message displayed from ApiError.message | accept | ApiError.message comes from the server's own `{ "error": "..." }` body — not from user input. Server is same-origin. Low risk |
| T-05-15 | Denial of Service | Race condition: fast range changes trigger overlapping fetches | mitigate | `cancelled` flag in useEffect cleanup prevents stale responses from overwriting newer data. Enforced by acceptance criteria: `grep "cancelled" returns match` |
| T-05-16 | Information Disclosure | Financial totals visible in React DevTools / browser memory | accept | Single-household, no auth by design. Data is the user's own wealth data — no privacy concern |
</threat_model>
