```typescript
import { useState } from 'react'
import { AreaChart, LineChart, BarChart2 } from 'lucide-react'

export type ChartType = 'area' | 'line' | 'bar'

// Exported hook so DashboardPage (and future ProjectionsPage) can own the state
// and pass both value and setter down as props.
export function useChartType(storageKey: string): [ChartType, (t: ChartType) => void] {
  const [chartType, setChartType] = useState<ChartType>(() => {
    const stored = localStorage.getItem(storageKey)
    // Validate stored value — only accept the three known values
    if (stored === 'area' || stored === 'line' || stored === 'bar') return stored
    return 'area'
  })

  function update(t: ChartType) {
    setChartType(t)
    localStorage.setItem(storageKey, t)
  }

  return [chartType, update]
}

// ── Component ──────────────────────────────────────────────────────────────────

interface ChartTypeSelectorProps {
  value: ChartType
  onChange: (t: ChartType) => void
}

export default function ChartTypeSelector({ value, onChange }: ChartTypeSelectorProps) {
  const buttons: { type: ChartType; icon: React.ReactNode; label: string }[] = [
    { type: 'area',  icon: <AreaChart size={16} />,  label: 'Stacked Area chart' },
    { type: 'line',  icon: <LineChart size={16} />,  label: 'Line chart' },
    { type: 'bar',   icon: <BarChart2 size={16} />,  label: 'Stacked Bar chart' },
  ]

  return (
    <div className="flex gap-1">
      {buttons.map(btn => (
        <button
          key={btn.type}
          aria-label={btn.label}
          onClick={() => onChange(btn.type)}
          className={
            value === btn.type
              ? 'p-2 rounded border border-blue-600 bg-blue-600 text-white'
              : 'p-2 rounded border border-gray-200 bg-white text-gray-500 hover:bg-gray-50 transition-colors'
          }
        >
          {btn.icon}
        </button>
      ))}
    </div>
  )
}
```

**CRITICAL RULES — all must be followed exactly:**

1. **dataKey MUST be `s.category_id` (slug) — NEVER `s.category_name`**. Category names can contain dots (e.g. "U.S. Stocks") which Recharts parses as nested object paths → undefined values. Slugs are safe strings without dots.

2. **`<div className="h-[420px]">` MUST wrap ResponsiveContainer**. Without explicit parent height, `ResponsiveContainer` computes 0px and renders a blank chart.

3. **All `<Area>` and `<Bar>` components MUST have `stackId="wealth"`** — stacking is disabled without a shared stackId.

4. **Total wealth `<Line>` MUST NOT have stackId** — it is an absolute value overlay, not stacked. It renders after all per-category series so it paints on top.

5. **Use `<ComposedChart>` — not `<AreaChart>`, `<LineChart>`, or `<BarChart>`**. ComposedChart is the only component that supports mixing Area + Line (for the total overlay) in a single chart.

**Data flattening** — each chart data point has shape:
`{ month: string, total: number, [category_id]: number }`


**Complete component:**
```typescript
import {
  ComposedChart, Area, Line, Bar,
  XAxis, YAxis, CartesianGrid, Tooltip, Legend,
  ResponsiveContainer,
} from 'recharts'
import type { SummaryResponse } from '../types/index'
import type { ChartType } from './ChartTypeSelector'

interface WealthChartProps {
  data: SummaryResponse
  chartType: ChartType
}

function buildChartData(response: SummaryResponse): Record<string, string | number>[] {
  return response.months.map((month, i) => {
    const point: Record<string, string | number> = { month, total: response.totals[i] }
    for (const s of response.series) {
      point[s.category_id] = s.values[i]
    }
    return point
  })
}

function CustomTooltip({ active, payload, label }: { active?: boolean; payload?: any[]; label?: string }) {
  if (!active || !payload?.length) return null
  const fmt = new Intl.NumberFormat('de-DE', { style: 'currency', currency: 'EUR' })
  const totalEntry = payload.find((p: any) => p.dataKey === 'total')
  const categories = payload.filter((p: any) => p.dataKey !== 'total')
  return (
    <div className="bg-white border border-gray-200 rounded-lg shadow-lg p-3 text-sm">
      <p className="font-medium text-gray-900 mb-2">{label}</p>
      {categories.map((p: any) => (
        <div key={p.dataKey} className="flex justify-between gap-4">
          <span className="font-medium" style={{ color: p.color }}>{p.name}</span>
          <span className="font-medium">{fmt.format(p.value)}</span>
        </div>
      ))}
      {totalEntry && (
        <div className="flex justify-between gap-4 font-bold border-t border-gray-200 mt-1 pt-1">
          <span>Total</span>
          <span>{fmt.format(totalEntry.value)}</span>
        </div>
      )}
    </div>
  )
}

export default function WealthChart({ data, chartType }: WealthChartProps) {
  const chartData = buildChartData(data)
  const { series } = data

  return (
    <div className="bg-white rounded-lg border border-gray-200 p-4">
      {/* MANDATORY: explicit height parent — ResponsiveContainer reads parent height */}
      <div className="h-[420px]">
        <ResponsiveContainer width="100%" height="100%">
          <ComposedChart data={chartData}>
            <CartesianGrid strokeDasharray="3 3" stroke="#E5E7EB" />
            <XAxis
              dataKey="month"
              tickFormatter={(m: string) => m.slice(0, 7)}
              tick={{ fontSize: 12, fill: '#6B7280' }}
            />
            <YAxis
              tickFormatter={(v: number) =>
                new Intl.NumberFormat('de-DE', { notation: 'compact', maximumFractionDigits: 0 }).format(v)
              }
              tick={{ fontSize: 12, fill: '#6B7280' }}
            />
            <Tooltip content={<CustomTooltip />} />
            <Legend wrapperStyle={{ fontSize: '12px', color: '#374151' }} />

            {/* Stacked Area mode — stackId required on EVERY Area */}
            {chartType === 'area' && series.map(s => (
              <Area
                key={s.category_id}
                dataKey={s.category_id}
                stackId="wealth"
                fill={s.color}
                stroke={s.color}
                fillOpacity={0.6}
                name={s.category_name}
              />
            ))}

            {/* Line mode — no stackId */}
            {chartType === 'line' && series.map(s => (
              <Line
                key={s.category_id}
                dataKey={s.category_id}
                stroke={s.color}
                dot={false}
                strokeWidth={2}
                name={s.category_name}
              />
            ))}

            {/* Stacked Bar mode — stackId required on EVERY Bar */}
            {chartType === 'bar' && series.map(s => (
              <Bar
                key={s.category_id}
                dataKey={s.category_id}
                stackId="wealth"
                fill={s.color}
                name={s.category_name}
              />
            ))}

            {/* Total wealth line — NO stackId ever; absolute overlay; renders last to paint on top */}
            <Line
              dataKey="total"
              stroke="#111827"
              strokeWidth={2}
              dot={false}
              name="Total"
            />
          </ComposedChart>
        </ResponsiveContainer>
      </div>
    </div>
  )
}
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-08 | XSS | CustomTooltip — `p.name` (category_name) rendered in JSX | mitigate | React JSX renders text nodes via `textContent`, not `innerHTML` — HTML special chars are escaped automatically. Never use `dangerouslySetInnerHTML`. Enforced by code pattern in this plan |
| T-05-09 | XSS | CustomTooltip — `p.color` in `style={{ color: p.color }}` | accept | CSS color values with XSS payloads (e.g. `expression(...)`) are no-ops in modern browsers. Recharts sanitises the stroke/fill props before SVG rendering |
| T-05-10 | Tampering | localStorage `dashboard-chart-type` | accept | Stored value is validated before use: `if (stored === 'area' \|\| stored === 'line' \|\| stored === 'bar') return stored` — any injected value falls through to default `'area'`. No privilege escalation possible |
