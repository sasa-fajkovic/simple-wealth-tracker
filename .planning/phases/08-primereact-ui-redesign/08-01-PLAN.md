Final `main.tsx`:
```typescript
import React from 'react'
import ReactDOM from 'react-dom/client'
import { BrowserRouter } from 'react-router-dom'
import { PrimeReactProvider } from 'primereact/api'
import 'primereact/resources/themes/lara-light-blue/theme.css'
import 'primereact/resources/primereact.css'
import 'primeicons/primeicons.css'
import './index.css'
import App from './App'

ReactDOM.createRoot(document.getElementById('root')!).render(
  <React.StrictMode>
    <PrimeReactProvider>
      <BrowserRouter>
        <App />
      </BrowserRouter>
    </PrimeReactProvider>
  </React.StrictMode>,
)
```

**CRITICAL — CSS import order is load-order-sensitive:** Theme must come first, then primereact.css, then primeicons, then index.css (Tailwind). Wrong order = broken PrimeReact component styles overridden by Tailwind base reset.

**Step 3 — Update `web/tailwind.config.ts`** by adding `important: true` at the top level (prevents Tailwind utility classes from losing specificity battles against PrimeReact theme):

```typescript
import type { Config } from 'tailwindcss'

export default {
  content: ['./index.html', './src/**/*.{ts,tsx}'],
  important: true,
  theme: {
    extend: {},
  },
  plugins: [],
} satisfies Config
```

Note: file is `.ts` not `.js`.

```typescript
export function getSummary(range: RangeKey, person?: string): Promise<SummaryResponse> {
  const params = new URLSearchParams({ range })
  if (person) params.set('person', person)
  return apiFetch(`/summary?${params.toString()}`)
}
```

```typescript
export function getPersons(): Promise<Person[]> {

  return apiFetch('/persons', { method: 'POST', body: JSON.stringify(data) })

}
```
