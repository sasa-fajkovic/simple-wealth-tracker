```tsx
import { useState, useEffect } from 'react'
import { Plus } from 'lucide-react'
import {
  getDataPoints, getAssets, createDataPoint, updateDataPoint, deleteDataPoint, ApiError,
} from '../../api/client'
import type { DataPoint, Asset, CreateDataPointPayload, UpdateDataPointPayload } from '../../types/index'
import DataPointModal from './DataPointModal'
import { DataTable, DataTableSortEvent } from 'primereact/datatable'
import { Column } from 'primereact/column'
import { Button } from 'primereact/button'
import { Message } from 'primereact/message'
import { Skeleton } from 'primereact/skeleton'
import { confirmDialog } from 'primereact/confirmdialog'

const eurFormatter = new Intl.NumberFormat('de-DE', { style: 'currency', currency: 'EUR' })

type ModalState = { mode: 'create' } | { mode: 'edit'; item: DataPoint }

export default function DataPointsTab() {
  const [rows, setRows] = useState<DataPoint[]>([])
  const [assets, setAssets] = useState<Asset[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const [retryCount, setRetryCount] = useState(0)
  const [sort, setSort] = useState<{ col: string; dir: 'asc' | 'desc' }>({ col: 'year_month', dir: 'desc' })
  const [modal, setModal] = useState<ModalState | null>(null)
  const [saving, setSaving] = useState(false)
  const [saveError, setSaveError] = useState<string | null>(null)

  useEffect(() => {
    let cancelled = false
    setLoading(true)
    setError(null)
    Promise.all([getDataPoints(), getAssets()])
      .then(([pts, ast]) => {
        if (!cancelled) { setRows(pts); setAssets(ast) }
      })
      .catch(err => {
        if (!cancelled) setError(err instanceof ApiError ? err.message : 'Unexpected error')
      })
      .finally(() => { if (!cancelled) setLoading(false) })
    return () => { cancelled = true }
  }, [retryCount])

  const assetMap = Object.fromEntries(assets.map(a => [a.id, a]))

  // Pre-flatten for DataTable column binding and sorting
  const displayRows = rows.map(row => ({
    ...row,
    assetName:    assetMap[row.asset_id]?.name ?? row.asset_id,
    categorySlug: assetMap[row.asset_id]?.category_id ?? '—',
    valueFormatted: eurFormatter.format(row.value),
  }))

  async function handleSave(payload: CreateDataPointPayload | UpdateDataPointPayload) {
    setSaving(true)
    setSaveError(null)
    try {
      if (modal?.mode === 'create') {
        await createDataPoint(payload as CreateDataPointPayload)
      } else if (modal?.mode === 'edit') {
        await updateDataPoint(modal.item.id, payload as UpdateDataPointPayload)
      }
      setModal(null)
      setRetryCount(c => c + 1)
    } catch (err) {
      setSaveError(err instanceof ApiError ? err.message : 'Unexpected error')
    } finally {
      setSaving(false)
    }
  }

  async function handleDelete(id: string) {
    await deleteDataPoint(id)
    setRetryCount(c => c + 1)
  }

  function openConfirm(id: string) {
    confirmDialog({
      message: 'Delete this data point? This cannot be undone.',
      header: 'Confirm Delete',
      icon: 'pi pi-exclamation-triangle',
      acceptClassName: 'p-button-danger',
      accept: () => handleDelete(id),
    })
  }

  const actionsTemplate = (row: typeof displayRows[number]) => (
    <div className="flex items-center gap-2">
      <Button
        icon="pi pi-pencil"
        text
        size="small"
        aria-label="Edit"
        onClick={() => {
          const original = rows.find(r => r.id === row.id)!
          setModal({ mode: 'edit', item: original })
        }}
      />
      <Button
        icon="pi pi-trash"
        text
        size="small"
        severity="danger"
        aria-label="Delete"
        onClick={() => openConfirm(row.id)}
      />
    </div>
  )

  return (
    <div>
      <div className="flex items-center justify-between mb-4">
        <h2 className="text-sm font-medium text-gray-700">Data Points</h2>
        <Button
          label="Add Data Point"
          icon="pi pi-plus"
          size="small"
          onClick={() => setModal({ mode: 'create' })}
        />
      </div>

      {error && (
        <div className="mb-4">
          <Message severity="error" text={`Could not load data: ${error}`} className="w-full" />
          <div className="mt-2">
            <Button label="Retry" link size="small" onClick={() => setRetryCount(c => c + 1)} />
          </div>
        </div>
      )}

      {loading && <Skeleton height="8rem" borderRadius="8px" />}

      {!loading && !error && (
        <DataTable
          value={displayRows}
          sortField={sort.col}
          sortOrder={sort.dir === 'asc' ? 1 : -1}
          onSort={(e: DataTableSortEvent) =>
            setSort({ col: e.sortField ?? sort.col, dir: e.sortOrder === 1 ? 'asc' : 'desc' })
          }
          emptyMessage="No data points yet. Click 'Add Data Point' to get started."
          stripedRows
          size="small"
        >
          <Column field="year_month"  header="Date"     sortable />
          <Column field="assetName"   header="Asset"    sortable />
          <Column field="categorySlug" header="Category" />
          <Column field="value"       header="Value"    sortable
            body={(row: typeof displayRows[number]) => row.valueFormatted} />
          <Column field="notes"       header="Notes"
            body={(row: typeof displayRows[number]) => row.notes ?? '—'} />
          <Column header="Actions"   body={actionsTemplate} style={{ width: '8rem' }} />
        </DataTable>
      )}

      {modal && (
        <DataPointModal
          mode={modal.mode}
          item={modal.mode === 'edit' ? modal.item : undefined}
          assets={assets}
          saving={saving}
          saveError={saveError}
          onSave={handleSave}
          onCancel={() => { setModal(null); setSaveError(null) }}
        />
      )}
    </div>
  )
}
```

### AssetsTab.tsx

Full rewrite. Keep all state and business logic. Replace `<table>` with `DataTable`. Add person fetching, personMap, and Person column.

Key additions over current implementation:
1. Import `getPersons` and `Person` type  
2. Add `persons` state
3. Extend `Promise.all` to 3-way: `[getAssets(), getCategories(), getPersons()]`
4. Build `personMap` derived constant
5. Pre-flatten rows to include `categoryName` and `personName` for DataTable
6. Add Person column (body template showing personMap lookup)
7. Pass `persons={persons}` to `AssetModal`

```tsx
import { useState, useEffect } from 'react'
import {
  getAssets, getCategories, getPersons,
  createAsset, updateAsset, deleteAsset, ApiError,
} from '../../api/client'
import type { Asset, Category, Person, CreateAssetPayload, UpdateAssetPayload } from '../../types/index'
import AssetModal from './AssetModal'
import { DataTable, DataTableSortEvent } from 'primereact/datatable'
import { Column } from 'primereact/column'
import { Button } from 'primereact/button'
import { Message } from 'primereact/message'
import { Skeleton } from 'primereact/skeleton'

type ModalState = { mode: 'create' } | { mode: 'edit'; item: Asset }

export default function AssetsTab() {
  const [rows, setRows] = useState<Asset[]>([])
  const [categories, setCategories] = useState<Category[]>([])
  const [persons, setPersons] = useState<Person[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const [retryCount, setRetryCount] = useState(0)
  const [sort, setSort] = useState<{ col: string; dir: 'asc' | 'desc' }>({ col: 'name', dir: 'asc' })
  const [modal, setModal] = useState<ModalState | null>(null)
  const [saving, setSaving] = useState(false)
  const [saveError, setSaveError] = useState<string | null>(null)
  const [deleteError, setDeleteError] = useState<{ id: string; message: string } | null>(null)

  useEffect(() => {
    let cancelled = false
    setLoading(true)
    setError(null)
    Promise.all([getAssets(), getCategories(), getPersons()])
      .then(([ast, cats, ppl]) => {
        if (!cancelled) { setRows(ast); setCategories(cats); setPersons(ppl) }
      })
      .catch(err => {
        if (!cancelled) setError(err instanceof ApiError ? err.message : 'Unexpected error')
      })
      .finally(() => { if (!cancelled) setLoading(false) })
    return () => { cancelled = true }
  }, [retryCount])

  const categoryMap = Object.fromEntries(categories.map(c => [c.id, c]))
  const personMap   = Object.fromEntries(persons.map(p => [p.id, p]))

  function formatGrowthRate(rate: number | null): string {
    return rate === null ? 'Inherits' : `${(rate * 100).toFixed(2)}%`
  }

  // Pre-flatten for DataTable binding and sorting
  const displayRows = rows.map(asset => ({
    ...asset,
    categoryName: categoryMap[asset.category_id]?.name ?? asset.category_id,
    personName:   personMap[asset.person_id ?? '']?.name ?? '—',
    growthFormatted: formatGrowthRate(asset.projected_yearly_growth),
  }))

  // Sort manually (computed fields not sortable by DataTable internally)
  const sortedRows = [...displayRows].sort((a, b) => {
    const mul = sort.dir === 'asc' ? 1 : -1
    if (sort.col === 'name')         return mul * a.name.localeCompare(b.name)
    if (sort.col === 'categoryName') return mul * a.categoryName.localeCompare(b.categoryName)
    if (sort.col === 'growth')       return mul * ((a.projected_yearly_growth ?? -Infinity) - (b.projected_yearly_growth ?? -Infinity))
    return 0
  })

  async function handleSave(payload: CreateAssetPayload | UpdateAssetPayload) {
    setSaving(true)
    setSaveError(null)
    try {
      if (modal?.mode === 'create') {
        await createAsset(payload as CreateAssetPayload)
      } else if (modal?.mode === 'edit') {
        await updateAsset(modal.item.id, payload as UpdateAssetPayload)
      }
      setModal(null)
      setRetryCount(c => c + 1)
    } catch (err) {
      setSaveError(err instanceof ApiError ? err.message : 'Unexpected error')
    } finally {
      setSaving(false)
    }
  }

  async function handleDelete(id: string) {
    setDeleteError(null)
    try {
      await deleteAsset(id)
      setRetryCount(c => c + 1)
    } catch (err) {
      if (err instanceof ApiError && err.status === 409) {
        setDeleteError({ id, message: err.message })
      }
    }
  }

  const actionsTemplate = (row: typeof sortedRows[number]) => (
    <div>
      <div className="flex items-center gap-2">
        <Button icon="pi pi-pencil" text size="small" aria-label="Edit"
          onClick={() => {
            const original = rows.find(r => r.id === row.id)!
            setModal({ mode: 'edit', item: original })
          }} />
        <Button icon="pi pi-trash" text size="small" severity="danger" aria-label="Delete"
          onClick={() => handleDelete(row.id)} />
      </div>
      {deleteError?.id === row.id && (
        <p className="text-xs text-red-600 mt-1">{deleteError.message}</p>
      )}
    </div>
  )

  return (
    <div>
      <div className="flex items-center justify-between mb-4">
        <h2 className="text-sm font-medium text-gray-700">Assets</h2>
        <Button label="Add Asset" icon="pi pi-plus" size="small"
          onClick={() => setModal({ mode: 'create' })} />
      </div>

      {error && (
        <div className="mb-4">
          <Message severity="error" text={`Could not load data: ${error}`} className="w-full" />
          <div className="mt-2">
            <Button label="Retry" link size="small" onClick={() => setRetryCount(c => c + 1)} />
          </div>
        </div>
      )}

      {loading && <Skeleton height="8rem" borderRadius="8px" />}

      {!loading && !error && (
        <DataTable
          value={sortedRows}
          sortField={sort.col}
          sortOrder={sort.dir === 'asc' ? 1 : -1}
          onSort={(e: DataTableSortEvent) =>
            setSort({ col: e.sortField ?? sort.col, dir: e.sortOrder === 1 ? 'asc' : 'desc' })
          }
          emptyMessage="No assets yet. Click 'Add Asset' to get started."
          stripedRows
          size="small"
        >
          <Column field="name"         header="Name"        sortable />
          <Column field="categoryName" header="Category"    sortable />
          <Column field="growth"       header="Growth Rate" sortable
            body={(row: typeof sortedRows[number]) => row.growthFormatted} />
          <Column field="location"     header="Location"
            body={(row: typeof sortedRows[number]) => row.location ?? '—'} />
          <Column field="notes"        header="Notes"
            body={(row: typeof sortedRows[number]) => row.notes ?? '—'} />
          <Column field="personName"   header="Person" />
          <Column header="Actions"     body={actionsTemplate} style={{ width: '8rem' }} />
        </DataTable>
      )}

      {modal && (
        <AssetModal
```

```tsx
// web/src/components/admin/PeopleTab.tsx
import { useState, useEffect } from 'react'
import {
  getPersons, createPerson, updatePerson, deletePerson, ApiError,
} from '../../api/client'
import type { Person, CreatePersonPayload, UpdatePersonPayload } from '../../types/index'
import PersonModal from './PersonModal'
import { DataTable, DataTableSortEvent } from 'primereact/datatable'
import { Column } from 'primereact/column'
import { Button } from 'primereact/button'
import { Message } from 'primereact/message'
import { Skeleton } from 'primereact/skeleton'

type ModalState = { mode: 'create' } | { mode: 'edit'; item: Person }

export default function PeopleTab() {
  const [rows, setRows] = useState<Person[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const [retryCount, setRetryCount] = useState(0)
  const [sort, setSort] = useState<{ col: string; dir: 'asc' | 'desc' }>({ col: 'name', dir: 'asc' })
  const [modal, setModal] = useState<ModalState | null>(null)
  const [saving, setSaving] = useState(false)
  const [saveError, setSaveError] = useState<string | null>(null)
  const [deleteError, setDeleteError] = useState<{ id: string; message: string } | null>(null)

  useEffect(() => {
    let cancelled = false
    setLoading(true)
    setError(null)
    getPersons()
      .then(ppl => { if (!cancelled) setRows(ppl) })
      .catch(err => {
        if (!cancelled) setError(err instanceof ApiError ? err.message : 'Unexpected error')
      })
      .finally(() => { if (!cancelled) setLoading(false) })
    return () => { cancelled = true }
  }, [retryCount])

  const sortedRows = [...rows].sort((a, b) => {
    const mul = sort.dir === 'asc' ? 1 : -1
    if (sort.col === 'name') return mul * a.name.localeCompare(b.name)
    if (sort.col === 'id')   return mul * a.id.localeCompare(b.id)
    return 0
  })

  async function handleSave(payload: CreatePersonPayload | UpdatePersonPayload) {
    setSaving(true)
    setSaveError(null)
    try {
      if (modal?.mode === 'create') {
        await createPerson(payload as CreatePersonPayload)
      } else if (modal?.mode === 'edit') {
        await updatePerson(modal.item.id, payload as UpdatePersonPayload)
      }
      setModal(null)
      setRetryCount(c => c + 1)
    } catch (err) {
      setSaveError(err instanceof ApiError ? err.message : 'Unexpected error')
    } finally {
      setSaving(false)
    }
  }

  async function handleDelete(id: string) {
    setDeleteError(null)
    try {
      await deletePerson(id)
      setRetryCount(c => c + 1)
    } catch (err) {
      if (err instanceof ApiError && err.status === 409) {
        setDeleteError({ id, message: err.message })
      }
    }
  }

  const actionsTemplate = (row: Person) => (
    <div>
      <div className="flex items-center gap-2">
        <Button icon="pi pi-pencil" text size="small" aria-label="Edit"
          onClick={() => setModal({ mode: 'edit', item: row })} />
        <Button icon="pi pi-trash" text size="small" severity="danger" aria-label="Delete"
          onClick={() => handleDelete(row.id)} />
      </div>
      {deleteError?.id === row.id && (
        <p className="text-xs text-red-600 mt-1">{deleteError.message}</p>
      )}
    </div>
  )

  return (
    <div>
      <div className="flex items-center justify-between mb-4">
        <h2 className="text-sm font-medium text-gray-700">People</h2>
        <Button label="Add Person" icon="pi pi-plus" size="small"
          onClick={() => setModal({ mode: 'create' })} />
      </div>

      {error && (
        <div className="mb-4">
          <Message severity="error" text={`Could not load persons: ${error}`} className="w-full" />
          <div className="mt-2">
            <Button label="Retry" link size="small" onClick={() => setRetryCount(c => c + 1)} />
          </div>
        </div>
      )}

      {loading && <Skeleton height="8rem" borderRadius="8px" />}

      {!loading && !error && (
        <DataTable
          value={sortedRows}
          sortField={sort.col}
          sortOrder={sort.dir === 'asc' ? 1 : -1}
          onSort={(e: DataTableSortEvent) =>
            setSort({ col: e.sortField ?? sort.col, dir: e.sortOrder === 1 ? 'asc' : 'desc' })
          }
          emptyMessage="No people yet. Click 'Add Person' to get started."
          stripedRows
          size="small"
        >
          <Column field="name" header="Name" sortable />
          <Column field="id"   header="Slug (ID)" sortable />
          <Column header="Actions" body={actionsTemplate} style={{ width: '8rem' }} />
        </DataTable>
      )}

      {modal && (
        <PersonModal
          mode={modal.mode}
          item={modal.mode === 'edit' ? modal.item : undefined}
          saving={saving}
          saveError={saveError}
          onSave={handleSave}
          onCancel={() => { setModal(null); setSaveError(null) }}
        />
      )}
    </div>
  )
}
```
