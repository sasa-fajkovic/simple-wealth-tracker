```tsx
// web/src/components/admin/PersonModal.tsx
import { useState } from 'react'
import type { Person, CreatePersonPayload, UpdatePersonPayload } from '../../types/index'
import { Dialog }    from 'primereact/dialog'
import { InputText } from 'primereact/inputtext'
import { Button }    from 'primereact/button'

// Mirrors server toSlug() exactly
function toSlug(name: string): string {
  return name.toLowerCase().replace(/\s+/g, '-').replace(/[^a-z0-9-]/g, '')
}

interface PersonModalProps {
  mode: 'create' | 'edit'
  item?: Person              // required when mode === 'edit'
  saving: boolean
  saveError: string | null
  onSave: (payload: CreatePersonPayload | UpdatePersonPayload) => void
  onCancel: () => void
}

export default function PersonModal({
  mode,
  item,
  saving,
  saveError,
  onSave,
  onCancel,
}: PersonModalProps) {
  const [name, setName] = useState(mode === 'edit' ? item!.name : '')
  const [validationError, setValidationError] = useState<string | null>(null)

  // Slug is always derived (create) or immutable (edit)
  const slugPreview = mode === 'create' ? toSlug(name) : item!.id

  function handleSubmit() {
    if (!name.trim()) {
      setValidationError('Name is required')
      return
    }
    setValidationError(null)
    onSave({ name: name.trim() })
  }

  return (
    <Dialog
      header={mode === 'create' ? 'Add Person' : 'Edit Person'}
      visible={true}
      onHide={onCancel}
      style={{ width: '28rem' }}
      modal
      draggable={false}
      footer={
        <div className="flex justify-end gap-2">
          <Button label="Cancel" outlined onClick={onCancel} type="button" />
          <Button
            label={saving ? 'Saving…' : mode === 'create' ? 'Add Person' : 'Save'}
            disabled={saving}
            onClick={handleSubmit}
          />
        </div>
      }
    >
      {/* Name */}
      <div className="mb-3">
        <label className="block text-xs font-medium text-gray-500 mb-1">Name</label>
        <InputText
          value={name}
          onChange={e => setName(e.target.value)}
          className="w-full"
          autoFocus
        />
      </div>

      {/* Slug preview — always read-only */}
      <div className="mb-4">
        <label className="block text-xs font-medium text-gray-500 mb-1">
          ID (slug) — read only
        </label>
        <InputText
          value={slugPreview}
          readOnly
          className="w-full bg-gray-50 cursor-not-allowed"
        />
      </div>

      {validationError && <p className="text-xs text-red-600 mb-2">{validationError}</p>}
      {saveError       && <p className="text-xs text-red-600 mb-2">{saveError}</p>}
    </Dialog>
  )
}
```

TYPOGRAPHY RULE: Only `font-medium` and `font-bold`. No `font-semibold` or `font-normal`.
SLUG RULE: Slug fields are ALWAYS `readOnly` in all modals — create mode shows live preview, edit mode shows immutable id.
