Full rewrite of `DashboardPage.tsx`. Keep all business logic; replace UI primitives with PrimeReact components.

```tsx
import { useState, useEffect } from 'react'
import { getSummary, getPersons, ApiError } from '../api/client'
import type { SummaryResponse, RangeKey, Person } from '../types/index'
import ChartTypeSelector, { useChartType } from '../components/ChartTypeSelector'
import WealthChart from '../components/WealthChart'
import SummaryCards from '../components/SummaryCards'
import { SelectButton, SelectButtonChangeEvent } from 'primereact/selectbutton'
import { Skeleton } from 'primereact/skeleton'
import { Message } from 'primereact/message'
import { Button } from 'primereact/button'
import { Card } from 'primereact/card'

const RANGES: { label: string; value: RangeKey }[] = [
  { label: 'YTD', value: 'ytd' }, { label: '6M',  value: '6m'  },
  { label: '1Y',  value: '1y'  }, { label: '2Y',  value: '2y'  },
  { label: '3Y',  value: '3y'  }, { label: '5Y',  value: '5y'  },
  { label: '10Y', value: '10y' }, { label: 'Max', value: 'max' },
]

export default function DashboardPage() {
  const [range, setRange] = useState<RangeKey>('1y')
  const [persons, setPersons] = useState<Person[]>([])
  const [person, setPerson] = useState<string | null>(null)  // null = "All"
  const [data, setData] = useState<SummaryResponse | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const [retryCount, setRetryCount] = useState(0)
  const [chartType, setChartType] = useChartType('dashboard-chart-type')

  // Page title — mount only
  useEffect(() => {
    document.title = 'Dashboard — WealthTrack'
  }, [])

  // Fetch persons once on mount — NOT tied to retryCount
  useEffect(() => {
    let cancelled = false
    getPersons()
      .then(list => { if (!cancelled) setPersons(list) })
      .catch(() => {}) // persons load silently; don't block dashboard
    return () => { cancelled = true }
  }, [])

  // Fetch summary whenever range, person, or retryCount changes
  useEffect(() => {
    let cancelled = false
    setLoading(true)
    setError(null)
    getSummary(range, person ?? undefined)
      .then(result => {
        if (!cancelled) setData(result)
      })
      .catch(err => {
        if (!cancelled) {
          setError(err instanceof ApiError ? err.message : 'Unexpected error')
        }
      })
      .finally(() => {
        if (!cancelled) setLoading(false)
      })
    return () => { cancelled = true }
  }, [range, person, retryCount])  // CRITICAL: person in dep array

  // Person SelectButton options: sentinel 'all' = null in state
  const personOptions = [
    { label: 'All', value: 'all' as const },
    ...persons.map(p => ({ label: p.name, value: p.id })),
  ]
  const personValue = person ?? 'all'

  return (
    <div className="min-h-screen bg-gray-50">
      <div className="max-w-7xl mx-auto px-4 sm:px-8 py-8">

        {/* Controls row: range SelectButton (left) + chart type selector (right) */}
        <div className="flex flex-wrap items-center justify-between gap-3 mb-4">
          <SelectButton
            value={range}
            onChange={(e: SelectButtonChangeEvent) => {
              if (e.value != null) setRange(e.value as RangeKey)
            }}
            options={RANGES}
            optionLabel="label"
            optionValue="value"
          />
          <ChartTypeSelector value={chartType} onChange={setChartType} />
        </div>

        {/* Person filter row — only shown when persons exist */}
        {persons.length > 0 && (
          <div className="mb-4">
            <SelectButton
              value={personValue}
              onChange={(e: SelectButtonChangeEvent) => {
                if (e.value != null) {
                  setPerson(e.value === 'all' ? null : e.value as string)
                }
              }}
              options={personOptions}
              optionLabel="label"
              optionValue="value"
            />
          </div>
        )}

        {/* Error banner */}
        {error && (
          <div className="mb-6">
            <Message
              severity="error"
              text="Could not load data. Check your connection."
              className="w-full"
            />
            <div className="mt-2">
              <Button
                label="Retry loading data"
                link
                onClick={() => setRetryCount(c => c + 1)}
              />
            </div>
          </div>
        )}

        {/* Loading skeleton */}
        {loading ? (
          <>
            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4 mb-6">
              {[0, 1, 2].map(i => (
                <Skeleton key={i} height="6rem" borderRadius="8px" />
              ))}
            </div>
            <Skeleton height="420px" borderRadius="8px" />
          </>
        ) : data ? (
          <>
            <SummaryCards data={data} />
            <Card className="mb-4">
              <WealthChart data={data} chartType={chartType} />
            </Card>
          </>
        ) : null}

      </div>
    </div>
  )
}
```

**Key rules to enforce:**
- Persons useEffect dep array MUST be `[]` — persons are not refetched on data retry
- Summary useEffect dep array MUST be `[range, person, retryCount]` — all three
- `getSummary(range, person ?? undefined)` — `null` maps to `undefined` to omit query param
- Person SelectButton uses sentinel value `'all'` to represent null state (avoids PrimeReact null-value quirks)
- `e.value != null` guard in both SelectButton onChange handlers prevents deselect events
- `font-semibold` and `font-normal` BANNED — use `font-medium` and `font-bold` only
