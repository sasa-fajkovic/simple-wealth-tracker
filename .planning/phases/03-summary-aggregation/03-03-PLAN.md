```typescript
// ── SummaryResponse ──────────────────────────────────────────────────────────

export interface SummaryResponse {
  months: string[]
  series: {
    category_id: string
    category_name: string
    color: string
    values: number[]
  }[]
  totals: number[]
  current_total: number
  period_delta_abs: number
  period_delta_pct: number
  category_breakdown: {
    category_id: string
    category_name: string
    color: string
    value: number
    pct_of_total: number
  }[]
}

/**
 * Aggregate LOCF-filled per-asset data into the SummaryResponse shape.
 *
 * Algorithm:
 * 1. For each category: sum across all assets in that category for each month
 *    → produces series[i].values[]
 * 2. totals[j] = sum of series[i].values[j] across all i (all categories)
 * 3. current_total = totals[last] (0 if months is empty)
 * 4. period_delta_abs = current_total - totals[0] (0 if months is empty)
 * 5. period_delta_pct = firstTotal === 0 ? 0 : (period_delta_abs / firstTotal) * 100
 * 6. category_breakdown[i].value = series[i].values[last]
 *    category_breakdown[i].pct_of_total = current_total === 0 ? 0 : (value / current_total) * 100
 *
 * SUM-04: returns months[], series[], totals[]
 * SUM-05: returns current_total, period_delta_abs, period_delta_pct, category_breakdown[]
 */
export function aggregateSummary(
  locfData: Map<string, Map<string, number>>,
  assets: Asset[],
  categories: Category[],
  months: string[]
): SummaryResponse {
  // Build per-category series: for each month, sum all assets in that category
  const series = categories.map((cat) => {
    const catAssets = assets.filter((a) => a.category_id === cat.id)
    const values = months.map((month) =>
      catAssets.reduce((sum, asset) => {
        return sum + (locfData.get(asset.id)?.get(month) ?? 0)
      }, 0)
    )
    return {
      category_id: cat.id,
      category_name: cat.name,
      color: cat.color,
      values,
    }
  })

  // totals[j] = sum across all series for month j
  const totals = months.map((_, j) =>
    series.reduce((sum, s) => sum + s.values[j], 0)
  )

  const currentTotal = totals.length > 0 ? totals[totals.length - 1] : 0
  const firstTotal = totals.length > 0 ? totals[0] : 0
  const periodDeltaAbs = totals.length > 0 ? currentTotal - firstTotal : 0
  // Divide-by-zero guard: if firstTotal is 0, delta% is 0 (not NaN or Infinity)
  const periodDeltaPct = firstTotal === 0 ? 0 : (periodDeltaAbs / firstTotal) * 100

  const categoryBreakdown = series.map((s) => {
    const value = s.values.length > 0 ? s.values[s.values.length - 1] : 0
    // Divide-by-zero guard: if currentTotal is 0, pct_of_total is 0
    const pctOfTotal = currentTotal === 0 ? 0 : (value / currentTotal) * 100
    return {
      category_id: s.category_id,
      category_name: s.category_name,
      color: s.color,
      value,
      pct_of_total: pctOfTotal,
    }
  })

  return {
    months,
    series,
    totals,
    current_total: currentTotal,
    period_delta_abs: periodDeltaAbs,
    period_delta_pct: periodDeltaPct,
    category_breakdown: categoryBreakdown,
  }
}
```

```typescript
// ── aggregateSummary ─────────────────────────────────────────────────────────

describe('aggregateSummary', () => {
  const cat1: Category = { id: 'stocks', name: 'Stocks', projected_yearly_growth: 0.08, color: '#6366f1' }
  const cat2: Category = { id: 'cash', name: 'Cash', projected_yearly_growth: 0.02, color: '#64748b' }

  test('period_delta_pct is 0 (not NaN/Infinity) when first total is 0', () => {
    // All assets have value 0 throughout — firstTotal = 0
    const locfData = new Map<string, Map<string, number>>()
    const a1 = makeAsset('a1')
    a1.category_id = 'stocks'
    const months = ['2024-01', '2024-02']
    locfData.set('a1', new Map([['2024-01', 0], ['2024-02', 500]]))
    const result = aggregateSummary(locfData, [a1], [cat1], months)
    assert.equal(result.period_delta_pct, 0, 'must be 0 when first total is 0, not Infinity')
    assert.ok(!isNaN(result.period_delta_pct), 'must not be NaN')
  })

  test('pct_of_total is 0 (not NaN) when current_total is 0', () => {
    const locfData = new Map<string, Map<string, number>>()
    const months = ['2024-01']
    // No assets → locfData is empty → all values are 0
    const result = aggregateSummary(locfData, [], [cat1], months)
    assert.equal(result.current_total, 0)
    // category_breakdown exists for cat1 but with pct_of_total = 0
    assert.equal(result.category_breakdown[0].pct_of_total, 0)
    assert.ok(!isNaN(result.category_breakdown[0].pct_of_total))
  })

  test('series[].values.length equals months.length for every category', () => {
    const months = ['2024-01', '2024-02', '2024-03']
    const locfData = new Map<string, Map<string, number>>()
    const a1 = makeAsset('a1')
    a1.category_id = 'stocks'
    locfData.set('a1', new Map([['2024-01', 100], ['2024-02', 200], ['2024-03', 300]]))
    const result = aggregateSummary(locfData, [a1], [cat1, cat2], months)
    assert.equal(result.series.length, 2, 'one series per category')
    for (const s of result.series) {
      assert.equal(s.values.length, months.length, `${s.category_id} values.length must match months.length`)
    }
  })

  test('totals equals per-month sum across all category series', () => {
    const months = ['2024-01', '2024-02']
    const locfData = new Map<string, Map<string, number>>()
    const a1 = makeAsset('a1')  // stocks
    const a2 = makeAsset('a2')  // cash
    a1.category_id = 'stocks'
    a2.category_id = 'cash'
    locfData.set('a1', new Map([['2024-01', 1000], ['2024-02', 1200]]))
    locfData.set('a2', new Map([['2024-01', 500], ['2024-02', 600]]))
    const result = aggregateSummary(locfData, [a1, a2], [cat1, cat2], months)
    assert.equal(result.totals[0], 1500, 'Jan total: 1000 + 500')
    assert.equal(result.totals[1], 1800, 'Feb total: 1200 + 600')
    assert.equal(result.current_total, 1800, 'current_total = last total')
  })

  test('period_delta_abs and period_delta_pct are correct', () => {
    const months = ['2024-01', '2024-02']
    const locfData = new Map<string, Map<string, number>>()
    const a1 = makeAsset('a1')
    a1.category_id = 'stocks'
    locfData.set('a1', new Map([['2024-01', 1000], ['2024-02', 1500]]))
    const result = aggregateSummary(locfData, [a1], [cat1], months)
    assert.equal(result.period_delta_abs, 500, '1500 - 1000 = 500')
    assert.ok(Math.abs(result.period_delta_pct - 50) < 0.001, '500/1000*100 = 50%')
  })

  test('category_breakdown pct_of_total values sum to ~100%', () => {
    const months = ['2024-01']
    const locfData = new Map<string, Map<string, number>>()
    const a1 = makeAsset('a1')
    const a2 = makeAsset('a2')
    a1.category_id = 'stocks'
    a2.category_id = 'cash'
    locfData.set('a1', new Map([['2024-01', 750]]))
    locfData.set('a2', new Map([['2024-01', 250]]))
    const result = aggregateSummary(locfData, [a1, a2], [cat1, cat2], months)
    const totalPct = result.category_breakdown.reduce((sum, b) => sum + b.pct_of_total, 0)
    assert.ok(Math.abs(totalPct - 100) < 0.001, `pct sum must be ~100, got ${totalPct}`)
  })

  test('empty database (no months) returns zero values without throwing', () => {
    const locfData = new Map<string, Map<string, number>>()
    const result = aggregateSummary(locfData, [], [cat1], [])
    assert.equal(result.months.length, 0)
    assert.equal(result.totals.length, 0)
    assert.equal(result.current_total, 0)
    assert.equal(result.period_delta_abs, 0)
    assert.equal(result.period_delta_pct, 0)
    assert.ok(!isNaN(result.period_delta_pct))
  })
})
```
