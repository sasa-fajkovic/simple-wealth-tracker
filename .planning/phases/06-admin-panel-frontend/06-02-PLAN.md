useEffect(() => {
  let cancelled = false
  setLoading(true)
  setError(null)
  Promise.all([getAssets(), getCategories()])
    .then(([ast, cats]) => {
      if (!cancelled) { setRows(ast); setCategories(cats) }
    })
    .catch(err => {
      if (!cancelled) setError(err instanceof ApiError ? err.message : 'Unexpected error')
    })
    .finally(() => { if (!cancelled) setLoading(false) })
  return () => { cancelled = true }
}, [retryCount])

**handleSave:**

async function handleSave(payload: CreateAssetPayload | UpdateAssetPayload) {
  setSaving(true)
  setSaveError(null)
  try {
    if (modal?.mode === 'create') {
      await createAsset(payload as CreateAssetPayload)
    } else if (modal?.mode === 'edit') {
      await updateAsset(modal.item.id, payload as UpdateAssetPayload)
    }
    setModal(null)
    setRetryCount(c => c + 1)
  } catch (err) {
    setSaveError(err instanceof ApiError ? err.message : 'Unexpected error')
  } finally {
    setSaving(false)
  }
}
