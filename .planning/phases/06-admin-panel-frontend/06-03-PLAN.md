```typescript
async function handleSave(payload: CreateCategoryPayload | UpdateCategoryPayload) {
  setSaving(true)
  setSaveError(null)
  try {
    if (modal?.mode === 'create') {
      await createCategory(payload as CreateCategoryPayload)
    } else if (modal?.mode === 'edit') {
      await updateCategory(modal.item.id, payload as UpdateCategoryPayload)
    }
    setModal(null)
    setRetryCount(c => c + 1)
  } catch (err) {
    setSaveError(err instanceof ApiError ? err.message : 'Unexpected error')
  } finally {
    setSaving(false)
  }
}
```
